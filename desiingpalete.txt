    <script>
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playRetroSound(type) {
            if (!audioCtx) return;
            if (audioCtx.state === 'suspended') {
                audioCtx.resume();
            }

            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            const now = audioCtx.currentTime;

            if (type === 'hover') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(320, now);
                osc.frequency.exponentialRampToValueAtTime(420, now + 0.03);
                gain.gain.setValueAtTime(0.005, now);
                gain.gain.linearRampToValueAtTime(0.0001, now + 0.03);
                osc.start(now);
                osc.stop(now + 0.03);
            } 
            else if (type === 'tab') {
                osc.type = 'triangle';
                osc.frequency.setValueAtTime(280, now);
                osc.frequency.exponentialRampToValueAtTime(450, now + 0.05);
                gain.gain.setValueAtTime(0.02, now);
                gain.gain.linearRampToValueAtTime(0.0001, now + 0.05);
                osc.start(now);
                osc.stop(now + 0.05);
            }
            else if (type === 'collect') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(440, now);
                osc.frequency.setValueAtTime(554.37, now + 0.04);
                osc.frequency.setValueAtTime(659.25, now + 0.08);
                gain.gain.setValueAtTime(0.015, now);
                gain.gain.linearRampToValueAtTime(0.015, now + 0.12);
                gain.gain.linearRampToValueAtTime(0.0001, now + 0.16);
                osc.start(now);
                osc.stop(now + 0.16);
            }
        }
    </script>

        // RENDERIZADO DEL INVENTARIO ADAPTABLE
        function renderInventory(filterCategory = "ALL", searchQuery = "") {
            const grid = document.getElementById('inventory-grid');
            const noResults = document.getElementById('no-results');
            grid.innerHTML = "";

            let filteredItems = itemsData;

            if (filterCategory !== "ALL") {
                filteredItems = filteredItems.filter(item => item.category === filterCategory);
            }

            if (searchQuery.trim() !== "") {
                const query = searchQuery.toLowerCase().trim();
                filteredItems = filteredItems.filter(item => 
                    item.name.toLowerCase().includes(query)
                );
            }

            if (filteredItems.length === 0) {
                noResults.classList.remove('hidden');
                noResults.classList.add('flex');
                grid.classList.add('hidden');
                return;
            } else {
                noResults.classList.add('hidden');
                noResults.classList.remove('flex');
                grid.classList.remove('hidden');
            }

            // Inyectar las tarjetas dinámicamente asegurándose que las proporciones usen el viewport height de la rejilla
            filteredItems.forEach(item => {
                const card = document.createElement('div');
                card.id = `item-card-${item.id}`;
                card.className = "item-card-flat px-3.5 py-1 sm:py-1.5 md:py-2.5 transition-all duration-100 cursor-pointer group";
                
                const formattedValue = typeof item.value === 'number' 
                    ? item.value.toLocaleString('es-ES', { minimumFractionDigits: item.value % 1 !== 0 ? 3 : 0 }) 
                    : item.value;

                card.innerHTML = `
                    <div class="flex items-center gap-2.5 min-w-0">
                        <!-- Icono vectorizado premium -->
                        <div class="w-6 h-6 sm:w-7 sm:h-7 rounded-full bg-black/40 flex items-center justify-center p-0.5 group-hover:scale-105 transition-transform shrink-0">
                            <svg class="w-5.5 h-5.5" viewBox="0 0 16 16" fill="none">
                                ${item.svg}
                            </svg>
                        </div>
                        <span class="text-[11px] sm:text-[12.5px] font-bold text-gray-400 group-hover:text-white transition-colors truncate">
                            ${item.name}
                        </span>
                    </div>
                    <span id="value-${item.id}" class="text-[11px] sm:text-[13px] font-black pl-1.5 shrink-0 transition-all duration-150" style="color: ${item.color};">
                        ${formattedValue}
                    </span>
                `;

                card.addEventListener('mouseenter', () => playRetroSound('hover'));
                
                card.addEventListener('click', () => {
                    playRetroSound('hover');
                    const increment = item.id === "LAVA" ? 0.05 : 1;
                    item.value = parseFloat((item.value + increment).toFixed(3));
                    updateItemUI(item.id);
                });

                grid.appendChild(card);
            });

            document.getElementById('active-count-label').textContent = `${itemsData.filter(i => i.value > 0).length}/${itemsData.length}`;
        }

        // Actualizar valores de recursos de forma individual
        function updateItemUI(itemId) {
            const item = itemsData.find(i => i.id === itemId);
            if (!item) return;

            const valueSpan = document.getElementById(`value-${itemId}`);
            if (valueSpan) {
                const formattedValue = typeof item.value === 'number' 
                    ? item.value.toLocaleString('es-ES', { minimumFractionDigits: item.value % 1 !== 0 ? 3 : 0 }) 
                    : item.value;
                
                valueSpan.textContent = formattedValue;
                
                valueSpan.classList.remove('flat-pulse-effect');
                void valueSpan.offsetWidth; // Forzar reflow CSS para reiniciar la animación
                valueSpan.classList.add('flat-pulse-effect');
            }

            document.getElementById('active-count-label').textContent = `${itemsData.filter(i => i.value > 0).length}/${itemsData.length}`;
        }

        // INICIALIZADORES DE EVENTOS
        document.addEventListener('DOMContentLoaded', () => {
            renderInventory();

            // Buscador
            const searchInput = document.getElementById('search-input');
            searchInput.addEventListener('input', (e) => {
                const activeTab = document.querySelector('.category-tab.active');
                const category = activeTab ? activeTab.getAttribute('data-category') : "ALL";
                renderInventory(category, e.target.value);
            });

            // Pestañas de Navegación
            const tabs = document.querySelectorAll('.category-tab');
            tabs.forEach(tab => {
                tab.addEventListener('click', () => {
                    playRetroSound('tab');
                    tabs.forEach(t => t.classList.remove('active'));
                    tab.classList.add('active');

                    const category = tab.getAttribute('data-category');
                    renderInventory(category, searchInput.value);
                });
            });

            // Acción: Simular Recolección (Añade valores masivos a elementos aleatorios)
            const collectBtn = document.getElementById('collect-btn');
            collectBtn.addEventListener('click', () => {
                playRetroSound('collect');
                
                const randomIndexes = [];
                while (randomIndexes.length < 5) {
                    const idx = Math.floor(Math.random() * itemsData.length);
                    if (!randomIndexes.includes(idx)) randomIndexes.push(idx);
                }

                randomIndexes.forEach(idx => {
                    const item = itemsData[idx];
                    let addedValue = 0;
                    
                    if (item.id === "SAND" || item.id === "COPPER") {
                        addedValue = Math.floor(Math.random() * 30000) + 5000;
                    } else if (item.id === "LAVA") {
                        addedValue = parseFloat((Math.random() * 0.08).toFixed(3));
                    } else {
                        addedValue = Math.floor(Math.random() * 15) + 3;
                    }

                    item.value = parseFloat((item.value + addedValue).toFixed(3));
                    updateItemUI(item.id);
                });
            });

            // Acción: Reiniciar a Cero
            const resetBtn = document.getElementById('reset-btn');
            resetBtn.addEventListener('click', () => {
                playRetroSound('tab');
                itemsData.forEach(item => {
                    item.value = 0;
                    updateItemUI(item.id);
                });
            });
        });
    </script>
